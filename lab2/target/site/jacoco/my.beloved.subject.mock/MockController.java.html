package my.beloved.subject.mock;

import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;
import java.lang.reflect.Proxy;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

class MockController {
    Map<Object, MockInvocationHandler> proxyToMock = new HashMap<>();
    Map<Object, ExpectInvocationHandler> proxyToExpected = new HashMap<>();
    
    public <T> T mock(Class<T> clazz) {
        if (!clazz.isInterface()) {
            // Throw
        }
        ClassLoader loader = clazz.getClassLoader();
        Class<?>[] interfaces = clazz.getInterfaces();
        
        var expectHandler = new ExpectInvocationHandler();
        Object expectProxy = Proxy.newProxyInstance(loader, interfaces, expectHandler);
        this.proxyToExpected.put(expectProxy, expectHandler);
        
        var mockHandler = new MockInvocationHandler(expectHandler);
        Object mockProxy = Proxy.newProxyInstance(loader, interfaces, mockHandler);
        this.proxyToMock.put(mockProxy, mockHandler);
        
        return clazz.cast(mockProxy);
    }

    // public MockCall expect() {

    // }
}

class ExpectInvocationHandler implements InvocationHandler {
    List<MockCall> expected = new ArrayList<>();

    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
        Object toReturn = method.getReturnType().getConstructor().newInstance();
        // toReturn.
        this.expected.add(new MockCall(method, args));
        return toReturn;
    }
}

class MockInvocationHandler implements InvocationHandler {
    ExpectInvocationHandler expectHandler;

    public MockInvocationHandler(ExpectInvocationHandler expectHandler) {
        this.expectHandler = expectHandler;
    }

    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
        return null;
    }
}
