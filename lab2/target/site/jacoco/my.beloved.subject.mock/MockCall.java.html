package my.beloved.subject.mock;

import java.lang.reflect.Method;

class MockCall {
    Method method;
    Object[] args;
    Object returns = null;
    int times = 1;
    MockCall after = null;

    public MockCall(Method method, Object[] args) {
        this.method = method;
        this.args = args;
    }

    public void returns(Object val) {
        this.returns = val;
    }

    public void times(int n) {
        this.times = n;
    }

    public void after(MockCall other) {
        this.after = other;
    }
}
